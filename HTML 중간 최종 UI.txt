    // ---------- Helper: aspect-ratio preserving draw (contain) ----------
    function drawContain(ctx, img, cw, ch){
      // 배경(레터박스) 먼저 칠하기
      ctx.fillStyle = '#0c0f14';
      ctx.fillRect(0, 0, cw, ch);

      const iw = img.width, ih = img.height;
      if(!iw || !ih) return;
      const scale = Math.min(cw / iw, ch / ih);
      const dw = iw * scale, dh = ih * scale;
      const dx = (cw - dw) / 2;
      const dy = (ch - dh) / 2;
      ctx.drawImage(img, dx, dy, dw, dh);
    }

    // ---------- Guided Grad-CAM (모의) 렌더러 ----------
    function renderGradDemo(){
      const g1 = document.getElementById('gradPreview');
      const g2 = document.getElementById('gradHeat');
      if(!g1 || !g2) return;

      // 사용할 이미지: previewImg(클라이언트) > serverImg(서버 응답)
      const previewEl = document.getElementById('previewImg');
      const serverEl  = document.getElementById('serverImg');
      const src = (previewEl && previewEl.src) ? previewEl.src : (serverEl && serverEl.src ? serverEl.src : null);

      const ctx1 = g1.getContext('2d');
      const ctx2 = g2.getContext('2d');
      ctx1.clearRect(0,0,g1.width,g1.height);
      ctx2.clearRect(0,0,g2.width,g2.height);

      if(!src){
        // 이미지가 없으면 placeholder만
        ctx1.fillStyle = '#0c0f14'; ctx1.fillRect(0,0,g1.width,g1.height);
        ctx2.fillStyle = '#0c0f14'; ctx2.fillRect(0,0,g2.width,g2.height);
        return;
      }

      const tmp = new Image();
      tmp.crossOrigin = 'anonymous';
      tmp.onload = ()=>{
        // 정사각형 캔버스에 비율유지로 그리기
        drawContain(ctx1, tmp, g1.width, g1.height);
        drawContain(ctx2, tmp, g2.width, g2.height);

        // Heat overlay (radial spots) — 이미지 위에 덧입힘
        const spots = [
          [g2.width*0.72, g2.height*0.28, Math.min(g2.width,g2.height)*0.20],
          [g2.width*0.38, g2.height*0.62, Math.min(g2.width,g2.height)*0.16],
          [g2.width*0.82, g2.height*0.74, Math.min(g2.width,g2.height)*0.12],
        ];
        spots.forEach(([cx, cy, r])=>{
          const rad = ctx2.createRadialGradient(cx, cy, r*0.05, cx, cy, r);
          rad.addColorStop(0.00,'rgba(255, 80, 60, 0.95)');
          rad.addColorStop(0.35,'rgba(255,200,60, 0.70)');
          rad.addColorStop(1.00,'rgba(255, 80, 60, 0.00)');
          ctx2.globalAlpha = 0.65;
          ctx2.fillStyle = rad;
          ctx2.beginPath();
          ctx2.rect(0,0,g2.width,g2.height);
          ctx2.fill();
          ctx2.globalAlpha = 1;
        });
      };
      tmp.src = src; // data:URL or base64 data URL
    }
